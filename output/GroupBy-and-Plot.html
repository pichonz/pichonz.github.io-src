def plot(group):
    mask = group['Volume'].apply(lambda x: x>1000)
    mask2 = group['Symbol'] == 'A'
    mask3 = group['Symbol'] == 'B'
    
    return group['Volume'][mask & (mask2 | mask3)].sum()

a = group.apply(plot)
a = pd.DataFrame(a)
a = a.rename(columns={0:'Volume'})
a.plot();
